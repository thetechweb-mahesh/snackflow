AuthController

Login
public function login(Request $request)
{
    $request->validate([
        'email'=>'required|email',
        'password'=>'required'
    ]);

    if (!Auth::attempt($request->only('email','password'))) {
        return response()->json([
            'message'=>'Invalid credentials'
        ],401);
    }

    $user = Auth::user();

    $token = $user
        ->createToken('snackflow')
        ->plainTextToken;

    return response()->json([
        'user'=>$user,
        'token'=>$token
    ]);
}
Route

    'login'
]);
